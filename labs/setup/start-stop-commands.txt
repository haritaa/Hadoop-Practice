START COMMANDS
===============
#Hadoop HDFS
start-dfs.sh

#YARN
start-yarn.sh

#JobHistoryServer
mr-jobhistory-daemon.sh start historyserver

#Hive Metastore
nohup hive --service metastore >/dev/null 2>&1 &

#Hive Server
nohup hive --service hiveserver2 >/dev/null 2>&1 &

#Hue Server
sudo nohup /opt/hue/build/env/bin/supervisor >/dev/null 2>&1 &

#Start HBase
start-hbase.sh

#Spark Master
start-master.sh

#Spark Worker
start-slave.sh <spark-master-url>

STOP COMMANDS
=============
#JobHistoryServer
mr-jobhistory-daemon.sh stop historyserver

#YARN
stop-yarn.sh

#Hadoop HDFS
stop-dfs.sh

#Hue Server
ps -ef | grep hue
kill -9 <hue-pid>

#Hive Server
ps -ef | grep hive
kill -9 <hiveserver-pid>

#Hive Metastore
ps -ef | grep hive
kill -9 <hivemetastore-pid>

#Stop HBase
stop-hbase.sh

#Spark Worker
stop-slaves.sh

#Spark Master
stop-master.sh
